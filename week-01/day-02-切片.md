"""  
                    不传指定参数时  
start    从哪开始切取决于step,step为正从索引0开始,为负从索引-1开始      包含起始索引  
stop     到哪停止切取决于step,step为正到末尾结束,为负到开头结束        不包含结束索引  
step     step为步长,默认为1,为2时的意思是切一次跳过2个元素且取第一个元素(隔位取)    可以为负值-1代表从后往前,方向的转变  
字符串、列表、元组都支持切片  
"""  
num = [1, 2, 3, 4, 5, 6, 7, 8]  
nums = []  
print(num[:])   # 都为默认值,结果是复制的效果  
print(num[-4:-1], num[-1:-4:-1])  # [-1:-4]如果不改step它默认为1,说明它没有调转方向,而开始又设定为-1,就切不了会自动修正到边界  
print(num[:3], num[-3:])  # 取开头3个元素      取末尾3个元素  
print(num[::-1])  # 反向取,把列表元素反转  
print(num[1:7:2])  # 从索引1开始切,到索引7结束但不包含7,步长为2 [2, 4, 6]  
  
# 切片插入  
nums[0:4] = [1, 2, 3, 4]  
nums[3:3] = [100]  # [1, 2, 3, 100, 4]  
# 切片替换  
nums[1:3] = [99, 88]  
# 切片删除  
nums[1:3] = []
